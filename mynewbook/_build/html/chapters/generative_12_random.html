Fokozatos kitöltés#

Véletlen számokat sokféleképp lehet használni. Ebben az esetben random.random() segítségével 0,1 közötti számot fogunk generálni és azt egy változóhoz viszonyítani, aminek az értékét fokozatosan növeljük.

Forrás: []

from PIL import Image, ImageDraw
import random

image_width = 800
image_height = 500
tile = 10

image = Image.new("RGB", (image_width, image_height))
draw = ImageDraw.Draw(image)

def resetImage(width, height, bg):
    draw.rectangle([0,0,width,height],fill=bg)

resetImage(image_width,image_height,"white")

threshold = 0.01

for i in range (0,image_width,tile):
    for j in range (0,image_height, tile):
        r = random.random()        
        if r < threshold:
            draw.rectangle([i,j,i+tile,j+tile],fill="red")
        else:
            draw.rectangle([i,j,i+tile,j+tile],fill="white")
    threshold+=0.01

display(image)
